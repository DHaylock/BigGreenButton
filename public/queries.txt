CREATE TABLE `BristolBigGreenButton`.`locations` ( `id` INT NOT NULL AUTO_INCREMENT , `lat` VARCHAR(50) NOT NULL , `lng` VARCHAR(50) NOT NULL , `timestamp` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `pledgeid` VARCHAR NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;


CREATE TABLE `BristolBigGreenButton`.`Pledges` ( `created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `pledgeid` VARCHAR(50) NOT NULL , `pledgename` VARCHAR(50) NOT NULL , `pledge` TEXT NOT NULL ) ENGINE = InnoDB;


SELECT a.lat, a.lng, a.timestamp, b.pledge FROM `locations` a join `Pledges` b WHERE a.pledgeid = b.pledgeid

SELECT a.pledgeid, a.lat, a.lng, a.timestamp, b.pledge FROM `locations` a join `Pledges` b WHERE a.pledgeid = b.pledgeid


SELECT a.pledgeid, a.lat, a.lng, a.timestamp, b.pledge,b.created_at,b.pledgename, IF (a.pledgeid = b.pledgeid,"true","false") as pledgeExists FROM `locations` a join `Pledges` b ORDER BY 'timestamp' ASC
